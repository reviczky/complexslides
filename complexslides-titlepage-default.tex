\startmodule complexslides-titlepage-default
\unprotect

% Style for title page
\definestyle[titlestyle][style=\bf\ssd, color=foreground:1]
\definestyle[institutestyle][style=\tfa, color=foreground:1]
\definestyle[authorstyle][style=\ssb, color=foreground:2]
\definestyle[datestyle][style=\tf, color=]
\definestyle[locationstyle][style=\tf, color=]

\startsetups[complexslides:titlepage]
  \startstandardmakeup
      \blank[force,2*big]
    \startalignment[middle]
      \titlestyle{\getvariable{talk}{title}}
      \blank[big]
      \authorstyle{\getvariable{talk}{author}}
      \blank[small]
      \institutestyle{\getvariable{talk}{institute}}
      \blank[big]
      \normal{\getvariable{talk}{thanks}}
      \vfill
      \locationstyle{\getvariable{talk}{location}}
      \blank[big]
    \stopalignment
    \setupinteraction
      [title=\getvariable{talk}{title}, subtitle=\getvariable{talk}{location}, author=\getvariable{talk}{author}]
  \stopstandardmakeup
\stopsetups

\protect
\stopmodule
